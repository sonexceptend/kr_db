SELECT t1.idw, t1.Фамилия,t1.Имя,t1.Отчество, t2.idv,t2.Вакансия,t1.Специальность,t1.id FROM (SELECT [tbWorkers].[id] as 'idw', [tbWorkers].[FirstName] as 'Фамилия',[tbWorkers].[SecondName] as 'Имя',[tbWorkers].[MiddleName] as 'Отчество',tbDoc.id_Vacancy,tbDoc.[Specialty] as 'Специальность',tbDoc.Id FROM [tbWorkers] FULL JOIN [tbDoctors] as tbDoc ON [tbWorkers].id = tbDoc.id_Workers) as t1 INNER JOIN (SELECT [tbVakancy].id as 'idv',[tbVakancy].Salary,[tbVakancy].Vacancy as 'Вакансия',[tbDoctors].Specialty,[tbDoctors].id FROM [tbVakancy] INNER JOIN [tbDoctors] ON  [tbDoctors].id_Vacancy = [tbVakancy].id) as t2 ON t1.id_Vacancy=t2.id
